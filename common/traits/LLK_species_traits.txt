trait_cor_dragon_totem = {
	cost = 0	# 特质点数花费
	icon = "gfx/interface/icons/traits/trait_cor_dragon_totem.dds"	# 图标路径
	allowed_archetypes = {
		BIOLOGICAL
		LITHOID
		ROBOT
		MACHINE
		PRESAPIENT

		WARSHIPGIRL
		SHENHAI
		STARSHIPGIRLS
		KANSEN
		INNOCENTSHOJO

		Magical_Bitch
		Magical_Girl
		PARASITE
		WHORE
		TENTACLE
		LOVEMACHINE
		YGWAWA
		HOLYROBOT
		MONSTER
		Lust_Girl
		Erotic_Creation
		##None
	}
	species_potential_add = {
		from = {
			AND = {
				has_technology = GE_tech_reactor
				OR = {
					has_civic = civic_cor_celestial_military
					has_civic = civic_cor_time_honored
				}
				has_country_flag = ge_leader_suigintou_get
				has_country_flag = ge_final_ascension_project_completed
			}
		}
	}
	species_possible_merge_add = {
		from = {
			AND = {
				has_technology = GE_tech_reactor
				OR = {
					has_civic = civic_cor_celestial_military
					has_civic = civic_cor_time_honored
				}
				has_country_flag = ge_leader_suigintou_get
				has_country_flag = ge_final_ascension_project_completed
			}
		}
	}
	species_possible_remove = {
		from = {
			AND = {
				has_technology = GE_tech_reactor
				has_country_flag = ge_final_ascension_project_completed
			}
		}
	}
	# 可使用该特质的原型
	initial = no	# 开局可用
	
	randomized = no	# 随机生成
	advanced_trait = yes

    #improves_leaders = yes

	#custom_tooltip = 
	modifier = {
		#  修正
        planet_jobs_produces_mult = 0.10
		#country_base_influence_produces_add = 1.0
		pop_happiness = 0.10                        #人口幸福度
		planet_stability_add = 10

	}
	potential_crossbreeding_chance = 0
	ai_weight = {   #AI的建造权重
        weight = 0  
        modifier = {    
            factor = 0
        }
	}	
}
