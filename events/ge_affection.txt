country_event = {
	id = ge_affection.11
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		every_owned_leader = {	#水银灯好感度后初始化
			limit = {
				has_leader_flag = ge_suigintou_flag
				NOT = {
					is_variable_set = ge_leader_affection
				}
			}
			set_variable = {
				which = ge_leader_affection
				value = 0
			}
		}
		every_owned_leader = { #雪华绮晶好感度后初始化
			limit = {
				has_leader_flag = roz_kirakishou_flag
				NOT = {
					is_variable_set = roz_kirakishou_affection
				}
			}
			set_variable = {
				which = roz_kirakishou_affection
				value = 60
			}
		}


		every_owned_leader = {	#水银灯-检查没有好感度特性的领袖并加特性
			if = {
				limit = {
					has_leader_flag = ge_suigintou_flag
					has_rozen_affection_trait = no
					# NOT = {
					# 	is_variable_set = ge_leader_affection
					# }
					check_variable = {
						which = ge_leader_affection
						value < 30
					}
					
				}
				switch = {
					trigger = leader_class
					# governor = {
					# 	add_trait = 
					# }
					scientist = {
						add_trait = leader_trait_ge_scientist_a_strangeness
					}
					commander = {
						add_trait = leader_trait_ge_admiral_a_strangeness
					}
				}
			}
		}
		every_owned_leader = {	#雪华绮晶-检查没有好感度特性的领袖并加特性
			if = {
				limit = {
					has_leader_flag = roz_kirakishou_flag
					NOR = {
						has_trait = leader_trait_krks_general_a_like
						has_trait = leader_trait_krks_general_a_love
						has_trait = leader_trait_krks_governor_a_like
						has_trait = leader_trait_krks_governor_a_love
					}
				}
				if = {
					limit = {
						check_variable = {
							which = ge_leader_affection
							value <= 90
						}
					}
					switch = {
						trigger = leader_class
						official = {
							add_trait = leader_trait_krks_governor_a_like
						}
						commander = {
							add_trait = leader_trait_krks_general_a_like
						}
					}
				}
				else_if = {
					limit = {
						check_variable = {
							which = ge_leader_affection
							value <= 90
						}
					}
					switch = {
						trigger = leader_class
						official = {
							add_trait = leader_trait_krks_governor_a_love
						}
						commander = {
							add_trait = leader_trait_krks_general_a_love
						}
					}
				}
			}
		}


		if = { #添加飞升特质（完成二段飞升）
			limit = {
				owner_main_species = {
					has_trait = trait_cor_dragon_totem
				}
			}
			every_owned_leader = {
				limit = {
					NOT= {
						has_trait = leader_trait_ge_ruler_bond
					}
				}
				add_trait = leader_trait_ge_ruler_bond
			}
			every_pool_leader = {
				limit = {
					NOT = {
						has_trait = leader_trait_ge_ruler_bond
					}
				}
				add_trait = leader_trait_ge_ruler_bond
			}
		}
	}
}

	id = ge_affection.15
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_country_flag = ge_leader_suigintou_get
	}
	immediate = {
		if = {
			limit = {
				fromfrom = {
					exists = leader
					leader = {
						has_leader_flag = ge_suigintou_flag
					}
				}
			}
			FromFromFrom = {
				save_event_target_as = roz_aft_enemy_fleet_target
				export_trigger_value_to_variable = {#敌方舰队舰船计数
					#居然不能prev.num_ships
					trigger = num_ships
					variable = roz_aft_num_ships_in_enemy_fleet
				}
				export_trigger_value_to_variable = {#敌方舰队战力计数
					trigger = fleet_power
					variable = roz_aft_enemy_fleet_power
				}
				if = {	#如果敌方舰队战力溢出为负数，则转为正数
					limit = { 
						check_variable = {
							which = roz_aft_enemy_fleet_power
							value < 0
						}
					}
					multiply_variable = {
						which = roz_aft_enemy_fleet_power
						value = -1
					}
				}
			}
			
			this = {
				change_variable = {#舰船计数转存至country域
					which = roz_aft_num_ships_in_enemy_fleet_check
					value = event_target:roz_aft_enemy_fleet_target.roz_aft_num_ships_in_enemy_fleet
				}
			}

			event_target:roz_aft_enemy_fleet_target = {#清除敌方舰队的舰船计数
				clear_variable = roz_aft_num_ships_in_enemy_fleet
			}
		}
	}
}

country_event = {
	###水银灯舰队司令赢得战斗加的好感值
	id = ge_affection.20
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_country_flag = ge_leader_suigintou_get
		fromfrom = {
			exists = leader
			leader = {
				
				has_leader_flag = ge_suigintou_flag
			}
		}
		NOT = {
			has_country_flag = roz_aft_fleet_combat_win_timed_flag
		}
	}
	immediate = {
		set_timed_country_flag = {
			flag = roz_aft_fleet_combat_win_timed_flag
			days = 2
		}
		fromfromfrom = {
			save_event_target_as = roz_aft_enemy_fleet_target
			export_trigger_value_to_variable = {
				trigger = num_ships
				variable = roz_aft_num_ships_in_enemy_fleet_end
			}
			
		}
		this = {
			subtract_variable = { #水银灯摧毁舰船数量计数
				which = roz_aft_num_ships_in_enemy_fleet_check
				value = root.fromfromfrom.roz_aft_num_ships_in_enemy_fleet_end
			}
		}
		fromfrom = {
			leader = {
				change_variable = { #摧毁敌舰数
					which = roz_aft_num_ships_destroy_value
					value = root.roz_aft_num_ships_in_enemy_fleet_check
				}
				divide_variable = { #倍率
					which = roz_aft_num_ships_destroy_value
					value = 400
				}
				change_variable = {
					which = ge_leader_aft_temp_number
					value = roz_aft_num_ships_destroy_value
				}
				clear_variable = roz_aft_num_ships_destroy_value


				change_variable = {#提取预先存储的战力计数
					which = ge_leader_aft_temp_number
					value = event_target:roz_aft_enemy_fleet_target.roz_aft_enemy_fleet_power
				}
				divide_variable = {#设置倍率
					which = ge_leader_aft_temp_number
					value = 20000
				}
				if = {
					limit = {
						check_variable = {#设置一次性获得的最大上限
							which = ge_leader_aft_temp_number
							value >= 1.2
						}
					}
					change_variable = {
						which = ge_leader_affection
						value = 1.2
					}
				}
				else_if = {
					limit = {
						check_variable = {
							which = ge_leader_aft_temp_number
							value < 1.2
						}
						check_variable = {
							which = ge_leader_aft_temp_number
							value > 0
						}
					}
					change_variable = {#根据战力水平增加好感度
						which = ge_leader_affection
						value = ge_leader_aft_temp_number
					}
				}
				else_if = {
					limit = {
						check_variable = {
							which = ge_leader_aft_temp_number
							value < 0
						}
					}
					change_variable = {#根据战力水平增加好感度
						which = ge_leader_affection
						value = 0.1
					}
				}
				clear_variable = ge_leader_aft_temp_number
			}
		}

		clear_variable = roz_aft_num_ships_in_enemy_fleet_check
		event_target:roz_aft_enemy_fleet_target = {
			clear_variable = roz_aft_num_ships_in_enemy_fleet_end
			clear_variable = roz_aft_enemy_fleet_power
			clear_variable = roz_aft_num_ships_in_enemy_fleet_check
		}
	}
