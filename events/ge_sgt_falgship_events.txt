				any_planet_within_border = {
					is_colonizable = no
					is_colony = no
					has_anomaly = no
				}
			}
			random_planet_within_border = {
				limit = {
					is_star = no
					is_colonizable = no
					is_colony = no
					has_anomaly = no

					category = roz_unknow_crashed_flagship_category
					target = root
				}
				save_event_target_as = roz_sgt_flagship_target
			}
		}
		else_if = {
			limit = {
				root = {
					is_ai = no
				}
			}
			random_planet_within_border = {
				limit = {
					is_star = no
				}
				add_anomaly = {
					category = roz_unknow_crashed_flagship_category
					target = solar_system.owner
				}
				save_event_target_as = roz_sgt_flagship_target
			}
		}
		create_fleet = {
			effect = {
				create_ship = {
					name = random
					random_existing_design = science
					graphical_culture = owner
					effect = {
						save_event_target_as = roz_sgt_flagship_science_target
					}
				}
				set_owner = prev
				set_location = event_target:roz_sgt_flagship_target
			}
		}
		event_target:roz_sgt_flagship_science_target = {
			ship_event = { id = ge_sgt_falgship_events.100 }
		}
	}
}
