        }

                }

        }
        else_if = {  #缺省
            limit = {
                OR = {
                    AND = { #只有中立

                        check_variable = {
                            which = roz_SK_invasion_dip_negative_ai
                            value = 0
                        }
                        check_variable = {
                            which = roz_SK_invasion_dip_aggressive_ai
                            value = 0
                        }

                        is_variable_set = roz_SK_invasion_dip_negative_ai
                        is_variable_set = roz_SK_invasion_dip_aggressive_ai

        }
